\section{小结}
\subsection{矩估计}
\begin{enumerate}
  \item 若总体的二阶矩存在那么，不论X服从何种分布， 则样本的平均值$\bar{x}$和二阶中心距$\hat{\delta}^2 = \frac{1}{n}\sum_{i=1}^n(x_i-\bar{x})^2$分布式总体均值和方差的矩估计。
  \item 均匀分布$U[\alpha, \beta]$, $E(X) = \frac{1}{2} (\alpha+\beta), Var(X) = \frac{1}{12}(\beta-\alpha)^2$
  \item if $\hat{\theta}$ 为$\theta$的矩估计，$q(\theta)$ 为 $\theta$的连续函数，那么$q(\hat{\theta})$也为$\theta$的矩估计。
\end{enumerate}
\subsection{极大似然估计}
\begin{enumerate}
  \item 多项分布的函数为：
      \begin{equation}\label{equation_multi_items1}
        P{N_1=n_1,N_2=n_2,...,N_m=n_m}=\frac{n!}{n_1!n_2!...n_m!}p_1^{n_1}...p_m^{n_m},\sum_{i=1}^m n_i= n, \sum_{i=1}^m p_i=1.
      \end{equation}
      $p_i$的极大似然估计为$\hat{p_i}=\frac{n_i}{n}$.
  \item 均匀分布$U[a,b]$的似然函数为
        \begin{equation}\label{equation_u_dist}
            L(a,b)=\left\{
                     \begin{array}{ll}
                       \frac{1}{(b-a)^n}, & {a\leq x\leq bs;} \\
                       0, & {others.}
                     \end{array}
                   \right.
        \end{equation}
        a,b的极大似然估计分别为$\hat{a}=x_{(1)},\hat{b}=x_{(n)}$.
  \item 二项分布$B(N,p)$的一阶矩为$E(X)=Np,Var(X)=Np(1-p)$.
\end{enumerate}

\subsection{均方误差准则}
\subsection{一致最小方差无偏估计}
\begin{enumerate}

  \item 均匀分布$U[0,\theta]$的一致最小方差无偏估计为$\frac{n+1}{n}x_{(n)}$
        
  \item $N(\mu,1)$的$\mu^2$的一致最小方差无偏估计达不到信息不等式的下界。
\end{enumerate}
