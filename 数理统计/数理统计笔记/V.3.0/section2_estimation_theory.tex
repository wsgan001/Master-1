\section{估计理论}

统计推断主要分为参数估计和假设检验，本章分析参数估计。参数估计又分为点估计和区间估计。
\subsection{参数的点估计}

\subsubsection{基本概念}
    \begin{itemize}
      \item 参数：在参数分布簇$\{ P_\theta ,\theta\in \Theta\}$ 中，通常将任何与总体有关的带估计的量都看成参数，它可以是决定总体参数和参数的实函数。可以表示为$q(\theta)$.
      \item 估计量：参数的任何实质统计量$T(X_1,X_2,...,X_n)$都称为估计量。$T(x_1,x_2,...,x_n)$称为估计值。
      \item 样本和样本观察值：$X_1,X_2,...,X_n$ 称为样本，$x_1,x_2,...,x_n$ 为样本观察值。在这里可以不区分。
      \item 统计量$T(X_1,X_2,...,X_n)$ 为参数$q(\theta)$的估计，记为$\hat{q}(x_1,x_2,...,x_n)=T(X_1,X_2,...,X_n)$.
    \end{itemize}

\subsubsection{频率替换法}

其中心思想是，\emph{大数定理，用频率替换相应的概率}。
考虑的对象是独立重复试验。

\begin{example}
 对于多项分布，$p(n1,n2,...,n_m)= n!/(n_1!n_2!...n_m!)p_1^{n_1}p_2^{n_2}...p_n^{n_m}$
  其最简单的估计是 $\hat{p}_i = n_i/n$.
\end{example}


$if \theta = (\theta_1,\theta_2,...,\theta_l) and p_i = h_i(\theta_1,\theta_2,...,\theta_l), i = 1,2,...,m,l\leq m$.
\begin{enumerate}
  \item 将 $\theta_j $表示为 $p_i$的函数.
  \item 带入 $q(\theta),$可得$q(\theta) = g(p1,p2,...pm)$.
  \item 然后 用频率替换概率$q(\theta) = g(n_{1}/n,n_{2}/n,...,n_{k}/n)$
\end{enumerate}

\subsubsection{矩估计法}

其中心思想是，\emph{大数定理,用样本矩替换总体矩}。样本矩依概率收敛于总体矩。依然使用替换原理。

由课本43页例题2.1.4可知，令$\mu _{1} = E(x)= \mu, \mu _2 = E(X^2) = Var(X)+(E(x))^2 = \sigma^2 + \mu^2$,将
$\mu$和$\sigma^2$表示为k阶中心距的函数即可。

\begin{description}
  \item[经验1] 矩估计不唯一，处理这种问题的一种简便方法是构造方程组的总体矩的阶数尽可能小。
  \item[经验2] $E(x)=\mu, Var(X)= E(X^2)-(EX)^2$.
  \item[经验3] 若总体的二阶矩存在那么，不论X服从何种分布， 则样本的平均值$\bar{x}$和二阶中心距$\hat{\delta}^2 = \frac{1}{n}\sum_{i=1}^n(x_i-\bar{x})^2$分布式总体均值和方差的矩估计。
\end{description}

\subsubsection{极大似然估计法}

中心思想是参数估计使得某事件发生的概率大。对每个x的值，取$p(x,\hat{\theta}(x)) = \max_{\theta \in \Theta}\{p(x,\theta)(x)\}$.
所以一个参数的极大似然估计也可以是一个实函数也可以是组合函数。

给定 ${x_1,x_2,...,x_n}$为简单样本，则样本的似然函数为：
\begin{equation}
    p(x_1,x_2,...,x_n;\theta) = \prod p(x_i,\theta)
\end{equation}

\begin{definition}
  若在参数空间$\Theta$ 中存在$\hat{\theta}(x_1,x_2,...,x_n)$使得下式\\
      \begin{center}
  $
    L(\hat{\theta}(x_1,x_2,...,x_n),x_1,x_2,...,x_n) = sup_{\theta \in \Theta}\{L(\theta,x_1,x_2,...,x_n)\}
  $\\
   \end{center}
  其中，$sup_{\theta \in \Theta}\{f(x)\}$表示$f(x)$在$\Theta$上的上确界。
\end{definition}

在求极大似然估计时的方法是对估计参数求偏导数，然后判断在参数空间内的取得上确界的参数值。
或者直接根据极大似然函数的单调性来判断即可。

\begin{description}
  \item[正态分布] $\hat{\mu} =  \frac{1}{n\sum_{i=1}^n x_i}$
  ,$\hat{\delta}^2 = \frac{1}{n\sum_{i=1}^n (x_i-\bar{x})^2}$.
  \item[正态分布的特例] 若已知$\mu\geq0$，那么$\hat{\mu}，\hat{\delta}^2$需要分情况讨论。\\
                        若$\bar{x}\geq0$,两者不变\\
                        若$\bar{x}\leq0$,$\hat{\mu}=0,\hat{\delta}^2 = 1/n\sum_{i=1}^n (x_i)^2$,由求偏导数易证。

  \item[Tips] 参数$\theta$的函数为$q(\theta)$,若极大似然估计$\hat{\theta}$存在，则$q(\hat{\theta})$为$q(\theta)$的极大似然估计。
\end{description}
\subsection{评价估计量的评优准则}
\subsubsection{均方误差准则}
\begin{definition}
  均方误差简记为MSE,即：
  \begin{equation}\label{equation_MSE}
    MSE_{\theta} = E_{\theta}(T(x)-q(\theta))^2
  \end{equation}

  若$MSE_{\theta}<\infty$,有：


   \begin{equation}\label{equation_MSE2}
     \begin{array}{ll}
       MSE_{\theta} = Var_{\theta}(T(x))+b^2(q(\theta),T); \\
       b^2(q(\theta),T)=E_{\theta}(T(x)-q(\theta)).
     \end{array}
    \end{equation}




\end{definition}

\begin{proposition}
  假设$T^{*}(x)$对任意$q(\theta)$的估计$T(x)$均方误差最小，这样的$T^{*}(x)$不存在。
\end{proposition}

\begin{description}
  \item[Tips] 在均方误差标准下，$\hat{\delta}^2$优于$S^2$.

\end{description}

\subsubsection{无偏估计}

\begin{definition}
  设统计量$T(x)$是参数的一个估计，若对所有的$\theta \in \Theta$,有：
  \begin{equation}\label{equation_Unbiased_Estimate}
    E_{\theta}(T(x))=q(x)
  \end{equation}
  成立，即偏差$b^2(q(\theta),T)=0$,则$T(x)$为$q(\theta)$的无偏估计(Unbiased Estimate).否则为有偏估计。
\end{definition}

\textcolor[rgb]{1.00,0.00,0.00}{\textbf{特别的无偏估计：}}
\begin{description}
  \item[Tips1] 无论总体服从什么分布，样本均值$\bar{x}$与样本方差$S^2$均为总体均值$\mu$和总体方差$\delta^2$的无偏估计。
  \item[Tips2] 无偏估计不一定优于有偏估计，例如$S^2$和$\hat{\delta}^2$.无偏估计可能是\textbf{非容许的}.
  \item[Tips4] 无偏估计可能不存在，参见例子【XXX】.
  \item[Tips5] 若无偏估计存在，无偏估计一般唯一。\\
  例如：对任意$m,(1\leq m <n)$,\\
  \begin{center}
    $T(x)=\frac{1}{2m}\sum_{i=1}^m x_i+\frac{1}{2(n-m)}\sum_{i=m+1}^{n}x_i$\\
  \end{center}
  \item[Tips6]在函数变换下，无偏性可能消失。
\end{description}

\begin{definition}
  若参数$q(\theta)$的无偏估计存在，则称$q(\theta)$ \emph{可估的}。
  \textbf{$U_q$无偏估计集合}：
  \begin{equation}\label{equation_Uq}
    U_q = \{
        T(x):E_{\theta}(T(x))=q(\theta),Var_{\theta}<\infty,\forall \theta \in \Theta
    \}
  \end{equation}

  无偏估计的均方误差即为无偏估计的方差。

\end{definition}
\subsubsection{一致最小方差无偏估计}

\begin{definition}
  \textbf{最小方差无偏估计(UMVUE)：} 若$T*(x)\in U_q$,对$\forall T(x) \in U_q$:
  \begin{equation}\label{equation_UMVUE}
    Var_{\theta}(T*(x))\leq Var_{\theta}(T(x))
  \end{equation}
\end{definition}

\begin{theorem}
  (存在性) $T(x)\in U_q$,$T(x)$为$q(\theta)$的一致最小方差无偏估计的充分必要条件是，对$\forall T_0(x) \in U_0$:
  \begin{equation}\label{equation_UMVUE_exist}
    E_{\theta}(T_0(x)T(x))=0,\forall \theta \in \Theta
  \end{equation}

\end{theorem}

\begin{corollary}
  若$q_1(\theta)$,$q_2(\theta)$的一致最小方差无偏估计为$T_1(x)$和$T_2(x)$,那么，$\forall a,b,aT_1(x)+bT_2(x)$为$aq_1(x)+bq_2(x)$的一致最小方差无偏估计。
\end{corollary}


\begin{theorem}
  (唯一性)设$q(\theta)$可估，且$S(x),T(x)$均为$q(\theta)$的UMUVE,那么$\forall \theta \in \Theta, P_{\theta}{T(x)=S(x)}=1$ ，即在概率1下，$q(\theta)$d的UMUVE唯一。
\end{theorem}
 \textcolor[rgb]{0.98,0.00,0.00}{【证明存在性和唯一性】} \\
 \textcolor[rgb]{0.98,0.00,0.00}{Rao-Blackwell定理和Lahmann-Scheffe定理}\\

 \textcolor[rgb]{0.98,0.00,0.00}{定义完全统计量和完全充分统计量}\\

\textcolor[rgb]{1.00,0.00,0.00}{ \emph{ 【以上是用于求UMUVE.】}}

\subsection{信息不等式}
\begin{itemize}
  \item 既然无偏估计的方差不是零，则必存在一个下界,这个下界到底是多少？\emph{(已由Cramer-Rao不等式（信息不等式)揭示）}
  \item 若UMVUE存在，那么它的方差是否可以达到这个下界？\emph{(不一定)}
\end{itemize}



\begin{definition}\label{Cramer-Rao}
  \textbf{Cramer-Rao正规族：}若总体分布族为$\{p(x,\theta),\theta \in \Theta\}$,其中$p(x,\theta)$为概率密度函数，$\Theta$为直线上某一开区间。若分布族满足条件：
  \begin{enumerate}
    \item 支持$A_{\theta}={x:p(x,\theta)>0}$与参数$\theta$无关，对$\forall x \in A_{\theta}$,在参数空间$\Theta$上的偏导数$\frac{\partial \ln p(x,\theta)}{\partial \theta}$存在。
    \item $\forall \theta \in \Theta,\forall E_{\theta}|T(x_1,x_2,...,x_n)|<+\infty$,有：
\begin{center}
  $
        \frac{\partial}{\partial\theta}\int_{-\infty}^{+\infty}T(x_1,x_2,...,x_n)p(x_1,x_2,...,x_n)dx_1...dx_n
        = \int_{-\infty}^{+\infty}...\int_{-\infty}^{+\infty}T(x_1,x_2,...,x_n)\frac{\partial}{\partial\theta}p(x_1,x_2,...,x_n)dx_1...dx_n
  $
\end{center}
  则称分布族$\{p(x,\theta),\theta \in \Theta\}$为Cramer-Rao正规族。
  \end{enumerate}
\end{definition}

\begin{definition}
  \textbf{Fisher信息量：}
\end{definition}



\begin{theorem}
  \textbf{信息不等式：}\textcolor[rgb]{1.00,0.00,0.00}{重要！:}分布族$\{p(x,\theta),\theta \in \Theta\}$为Cramer-Rao正规族，且$0<I(\theta)<+\infty$,$\forall \theta,Var_{\theta}(T(x_1,x_2,...,x_n))<\infty$，令$\varphi(\theta)=E_{\theta}(T(x_1,x_2,...,x_n))$,则对$\forall \theta \in \Theta,\varphi(\theta)$可微，且
\begin{equation}\label{equation_information}
    Var_\theta(T(x_1,x_2,...,x_n))\geq \frac{(\varphi^{'}(\theta))^2}{nI(\theta)}
\end{equation}


\end{theorem}
由此可得出无偏估计方差的下界。

\subsection{相合估计}
\begin{definition}
    \textbf{相合估计：}设$\hat{q}_n = \hat{q}_n(x_1,x_2,...,x_n)$是参数$q(\theta)$的任一估计序列，如果$\{\hat{q_n}\}$依概率收敛于参数真值$q(\theta)$,即对任意$\xi>0$,有
    \begin{equation}\label{equation_suit}
        \lim_{n\rightarrow\infty}P\{|\hat{q}_n-q(\theta)|>\xi\}=0;
    \end{equation}
    则称$\hat{q}_n(x_1,x_2,...,x_n)$是参数$q(\theta)$的相合估计。
\end{definition}

相合估计反映了向本容量n越来越大时，估计量的性质，即所谓的大样本性质，当样本容量有限时，相合性无意义。

\begin{theorem}
    如果$\hat{q}_n(x_1,x_2,...,x_n)$是参数$q(\theta)$的相合估计，且函数$g(y)$在$y=q(\theta)$处连续，则$g(\hat{q}_n(x_1,x_2,...,x_n))$是参数$g(q(\theta))$的相合估计.
\end{theorem}

\textcolor[rgb]{1.00,0.00,0.00}{TIPS, 对于存在多个相合估计的情况，可与通过比较其渐进分布来比较其渐近方差的大小，最常用的渐进方差是正态分布。}
